**Recommended execution path** (one of two equivalent options — pick whichever the executor finds clearer; either produces identical bytes):

Option A — one-shot Go re-indent via `go run`:
```bash
cat > /tmp/reindent.go <<'EOF'
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "os"
)

func main() {
    if len(os.Args) < 2 {
        fmt.Fprintln(os.Stderr, "usage: reindent <file>")
        os.Exit(2)
    }
    path := os.Args[1]
    raw, err := os.ReadFile(path)
    if err != nil {
        fmt.Fprintln(os.Stderr, err)
        os.Exit(1)
    }
    var pretty bytes.Buffer
    if err := json.Indent(&pretty, raw, "", "    "); err != nil {
        fmt.Fprintln(os.Stderr, err)
        os.Exit(1)
    }
    pretty.WriteByte('\n')
    if err := os.WriteFile(path, pretty.Bytes(), 0o644); err != nil {
        fmt.Fprintln(os.Stderr, err)
        os.Exit(1)
    }
}
EOF
```
