<objective>
Close WR-01: `Holiday.Range()` yielding `h.StartDate` verbatim on the first iteration so a hand-built Holiday with a non-UTC StartDate produces a non-UTC-midnight first yield, contradicting the explicit godoc contract.

Purpose: Repairs HELP-04 (Range() helper). The defect is a defense-in-depth violation, not a production-callers bug — endpoint-returned Holidays already pass through `validateHolidays` which enforces UTC-midnight on Start/End. But the godoc IS the contract per CL-11 ("Every yielded Date is rebuilt via NewDate, so each yielded value is at UTC midnight regardless of the receiver's internal time.Time location"), and the implementation must satisfy what the contract claims.

Output: One-line behavior change in `holiday.go` (`d := h.StartDate` → `d := NewDate(h.StartDate.Year(), h.StartDate.Month(), h.StartDate.Day())`) plus one new subtest in `holiday_test.go` that constructs a non-UTC-StartDate Holiday and asserts the first yielded Date is UTC-midnight. The existing 5 TestHoliday_Range subtests are unchanged.
</objective>

func (h Holiday) Range() iter.Seq[Date] {
    return func(yield func(Date) bool) {
        if h.EndDate.Before(h.StartDate) {
            return
        }
        d := h.StartDate           // <-- WR-01: first yield is verbatim, not normalized
        for {
            if !yield(d) {
                return
            }
            if !d.Before(h.EndDate) {
                return
            }
            next := d.AddDate(0, 0, 1)
            d = NewDate(next.Year(), next.Month(), next.Day())   // <-- subsequent yields ARE normalized
        }
    }
}
```

t.Run("non-UTC StartDate yields UTC-midnight first Date (WR-01 regression)", func(t *testing.T) {
    t.Parallel()
    // Hand-build a Holiday with a non-UTC StartDate. Endpoint-returned
    // Holidays would never look like this (validateHolidays canonicalizes
    // to UTC-midnight), but the godoc contract on Range() explicitly
    // promises every yield is normalized through NewDate. WR-01 was the
    // first-iteration drift between the contract and the code.
    cet := time.FixedZone("CET", 3600)
    h := Holiday{
        StartDate: Date{Time: time.Date(2025, time.January, 18, 0, 0, 0, 0, cet)},
        EndDate:   NewDate(2025, time.January, 18),
    }
    var dates []Date
    for d := range h.Range() {
        dates = append(dates, d)
    }
    require.Len(t, dates, 1, "single-day span must yield exactly one Date")
    first := dates[0]
    assert.Equal(t, time.UTC, first.Location(),
        "first yielded Date must be UTC-midnight regardless of StartDate location, got %s", first.Location())
    assert.Equal(t, 0, first.Hour())
    assert.Equal(t, 0, first.Minute())
    assert.Equal(t, 0, first.Second())
    assert.Equal(t, 0, first.Nanosecond())
    assert.Equal(t, 2025, first.Year())
    assert.Equal(t, time.January, first.Month())
    assert.Equal(t, 18, first.Day())
})
```
